**morris算法**

```python
import numpy as np
import math


class Counter(object):
    def __init__(self):
        self.X = 0

    def morris_count(self):
        if np.random.rand() < math.pow(1 / 2, self.X):
            self.X = self.X + 1
        return math.pow(2, self.X) - 1


if __name__ == "__main__":
    counter = Counter()
    for i in range(100):
        C = counter.morris_count()
        print(C)
```

**morris+算法**

```python
import numpy as np
import math


class Counter(object):
    def __init__(self):
        self.n = 10000
        self.X = np.zeros(self.n)

    def morris_count(self):
        for i in range(self.n):
            if np.random.rand() < math.pow(1 / 2, self.X[i]):
                self.X[i] = self.X[i] + 1
        count =0
        for i in range(self.n):
            count = count + math.pow(2, self.X[i]) - 1
        return count/self.n


if __name__ == "__main__":
    counter = Counter()
    for i in range(100):
        C = counter.morris_count()
        print(C)
```
